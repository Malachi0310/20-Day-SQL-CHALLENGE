-- 1. Create the table structure
CREATE TABLE Employees (
    EmployeeID INT PRIMARY KEY,
    Name VARCHAR(100),
    Department VARCHAR(50),
    Salary INT,
    HireDate DATE,
    PerformanceRating INT
);
