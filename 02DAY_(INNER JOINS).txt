-- 1. Customers Table
CREATE TABLE Customers (
    CustomerID INT PRIMARY KEY,
    FirstName VARCHAR(50),
    City VARCHAR(50),
    MembershipLevel VARCHAR(20) -- 'Gold', 'Silver', 'Bronze'
);

-- 2. Orders Table
CREATE TABLE Orders (
    OrderID INT PRIMARY KEY,
    CustomerID INT,
    OrderDate DATE,
    TotalAmount DECIMAL(10,2)
);

-- 3. Products Table (The items sold in those orders)
CREATE TABLE Products (
    ProductID INT PRIMARY KEY,
    ProductName VARCHAR(100),
    Category VARCHAR(50),
    Price DECIMAL(10,2)
);

-- Write a query to show the FirstName of the customer and the TotalAmount of every order they placed.
SELECT c.FirstName, o.TotalAmount 
FROM Customers AS c 
INNER JOIN Orders AS o ON c.CustomerID = o.CustomerID;

-- How much revenue (SUM of TotalAmount) has the company made from customers living in 'New York'?
SELECT c.City, SUM(o.TotalAmount)
FROM Customers AS c
INNER JOIN Orders AS o ON c.CustomerID = o.CustomerID
WHERE c.City = "New York";

-- Find the FirstName and the total number of orders placed by each customer. (Hint: Use COUNT(OrderID) and GROUP BY
SELECT c.FirstName, COUNT(o.OrderID) AS OrdersAmt
FROM Customers AS c 
INNER JOIN Orders AS o ON c.CustomerID = o.CustomerID
GROUP BY c.CustomerID;

-- Show the FirstName of customers who have spent a total of more than $1,000 across all their orders combined
SELECT c.FirstName, SUM(o.TotalAmount)
FROM Customers AS c 
INNER JOIN Orders AS o ON c.CustomerID = o.CustomerID
GROUP BY c.CustomerID
HAVING SUM(o.TotalAmount) >= 1000;
